SELECT COUNT(*),TAB1.Score FROM
(SELECT *,Switch(
        Bramki_zdobyte - Bramki_stracone > 0, "Z",
        Bramki_zdobyte - Bramki_stracone < 0, "P",
        Bramki_zdobyte - Bramki_stracone = 0, "R",
        ) AS Score
FROM wyniki 
WHERE Gdzie like "W") TAB1
GROUP BY TAB1.Score;
